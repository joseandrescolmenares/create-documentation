### Chat Activities Definition

The utils/chatActivity.js file defines the available activities, each containing an ID, a name, a prompt for the language model, and an image URL.

**Example:**

```javascript
const prompts = require("./prompts");

const chatactivity = [
  { id: 1, name: "anger", prompt: prompts.anger, image: "https://media.istockphoto.com/id/1399584106/es/foto/primer-plano-de-una-adorable-ni%C3%B1a-de-pie-con-los-brazos-cruzados-y-luciendo-molesta-mientras.jpg?s=1024x1024&w=is&k=20&c=SSjMtLJQf1pItPaih4yTJZP6jpSIi0PCgAu_zp65KSU=" },
  { id: 2, name: "sad", prompt: prompts.sad,image:"https://cdn.pixabay.com/photo/2017/08/10/03/47/guy-2617866_1280.jpg" },
  { id: 3, name: "anxiety", prompt: prompts.anxiety,image:"https://media.istockphoto.com/id/1480254699/es/foto/mujer-estudiante-fatiga-con-dolor-de-cabeza-y-agotamiento-universitario-estr%C3%A9s-por-la-fecha.jpg?s=1024x1024&w=is&k=20&c=pBeAF4p82XWsYxq-RTzTV4vcjiKuqlJvpnfuidZvkEQ=" },
];

module.exports = chatactivity;
```
